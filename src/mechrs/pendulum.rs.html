1//! Nonlinear pendulum solved with RK4.
2//!
3//! Compares exact `θ'' = -(g/L)sin(θ)` against the small-angle approximation `θ'' = -(g/L)θ`.
4
5const G: f64 = 9.81;
6
7/// A simple pendulum of given length.
8pub struct Pendulum {
9    pub length: f64,
10    pub g: f64,
11}
12
13impl Pendulum {
14    /// Create a pendulum with `g = 9.81 m/s²`.
15    pub fn new(length: f64) -> Self {
16        Self { length, g: G }
17    }
18
19    /// Full nonlinear RK4 iterator: `θ'' = -(g/L)sin(θ)`.
20    /// Yields `(time, angle_rad)`.
21    pub fn states(&self, theta0_rad: f64, dt: f64) -> impl Iterator<Item = (f64, f64)> {
22        let gl = self.g / self.length;
23        std::iter::successors(Some((0.0_f64, theta0_rad, 0.0_f64)), move |&(t, th, w)| {
24            let f = |_t: f64, th: f64, w: f64| -> (f64, f64) { (w, -gl * th.sin()) };
25            let (dth1, dw1) = f(t, th, w);
26            let (dth2, dw2) = f(t + dt / 2.0, th + dt / 2.0 * dth1, w + dt / 2.0 * dw1);
27            let (dth3, dw3) = f(t + dt / 2.0, th + dt / 2.0 * dth2, w + dt / 2.0 * dw2);
28            let (dth4, dw4) = f(t + dt, th + dt * dth3, w + dt * dw3);
29            let th_next = th + (dt / 6.0) * (dth1 + 2.0 * dth2 + 2.0 * dth3 + dth4);
30            let w_next = w + (dt / 6.0) * (dw1 + 2.0 * dw2 + 2.0 * dw3 + dw4);
31            Some((t + dt, th_next, w_next))
32        })
33        .map(|(t, th, _)| (t, th))
34    }
35
36    /// Small-angle approximation RK4 iterator: `θ'' = -(g/L)θ`.
37    pub fn small_angle_states(&self, theta0_rad: f64, dt: f64) -> impl Iterator<Item = (f64, f64)> {
38        let gl = self.g / self.length;
39        std::iter::successors(Some((0.0_f64, theta0_rad, 0.0_f64)), move |&(t, th, w)| {
40            let f = |_t: f64, th: f64, w: f64| -> (f64, f64) { (w, -gl * th) };
41            let (dth1, dw1) = f(t, th, w);
42            let (dth2, dw2) = f(t + dt / 2.0, th + dt / 2.0 * dth1, w + dt / 2.0 * dw1);
43            let (dth3, dw3) = f(t + dt / 2.0, th + dt / 2.0 * dth2, w + dt / 2.0 * dw2);
44            let (dth4, dw4) = f(t + dt, th + dt * dth3, w + dt * dw3);
45            let th_next = th + (dt / 6.0) * (dth1 + 2.0 * dth2 + 2.0 * dth3 + dth4);
46            let w_next = w + (dt / 6.0) * (dw1 + 2.0 * dw2 + 2.0 * dw3 + dw4);
47            Some((t + dt, th_next, w_next))
48        })
49        .map(|(t, th, _)| (t, th))
50    }
51
52    /// Analytical period (small-angle only): `T = 2π√(L/g)`.
53    pub fn analytical_period(&self) -> f64 {
54        2.0 * std::f64::consts::PI * (self.length / self.g).sqrt()
55    }
56
57    /// Numerical period via first downward zero-crossing (time × 2).
58    pub fn numerical_period(&self, theta0_rad: f64, dt: f64) -> f64 {
59        let mut states = self.states(theta0_rad, dt);
60        let mut prev = states.next().unwrap_or((0.0, 0.0));
61        for curr in states {
62            if prev.1 > 0.0 && curr.1 <= 0.0 {
63                // Starting at theta0 > 0, omega = 0: first downward zero crossing is at T/4
64                return prev.0 * 4.0;
65            }
66            prev = curr;
67        }
68        f64::NAN
69    }
70}

72#[cfg(test)]
73mod tests {
74    use super::*;
75
76    #[test]
77    fn small_angle_period_matches_analytical() {
78        let p = Pendulum::new(1.0);
79        let numerical = p.numerical_period(0.05_f64.to_radians(), 0.0001);
80        approx::assert_abs_diff_eq!(numerical, p.analytical_period(), epsilon = 0.001);
81    }
82
83    #[test]
84    fn large_angle_period_longer_than_small_angle() {
85        let p = Pendulum::new(1.0);
86        let large = p.numerical_period(80.0_f64.to_radians(), 0.0001);
87        assert!(
88            large > p.analytical_period(),
89            "Large angle period ({:.4}) should exceed small-angle ({:.4})",
90            large,
91            p.analytical_period()
92        );
93    }
94
95    #[test]
96    fn nonlinear_vs_small_angle_diverge_at_large_amplitude() {
97        let p = Pendulum::new(1.0);
98        let theta0 = 60.0_f64.to_radians();
99        let dt = 0.001;
100        let steps = 500;
101
102        let nonlinear: Vec<f64> = p.states(theta0, dt).take(steps).map(|(_, th)| th).collect();
103        let small_angle: Vec<f64> = p
104            .small_angle_states(theta0, dt)
105            .take(steps)
106            .map(|(_, th)| th)
107            .collect();
108
109        let max_diff = nonlinear
110            .iter()
111            .zip(small_angle.iter())
112            .map(|(a, b)| (a - b).abs())
113            .fold(0.0_f64, f64::max);
114
115        assert!(
116            max_diff > 0.01,
117            "60° amplitude should show measurable divergence: {max_diff}"
118        );
119    }
