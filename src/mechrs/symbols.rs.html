1//! Symbol table with lifetime-bounded expression references.
2//!
3//! `Expr<'sym>` borrows its name from the owning `SymbolTable` — it cannot outlive it.
4//! This is enforced at compile time with no runtime cost.
5
6use std::collections::HashMap;
7
8/// A table of named physical quantities.
9pub struct SymbolTable {
10    vars: HashMap<String, f64>,
11}
12
13/// An expression borrowed from a [`SymbolTable`].
14///
15/// The lifetime `'sym` ensures this cannot outlive the table it was created from.
16pub struct Expr<'sym> {
17    /// Name borrowed (zero-copy) from the symbol table.
18    pub name: &'sym str,
19    pub value: f64,
20    pub unit: &'static str,
21}
22
23impl SymbolTable {
24    /// Create an empty symbol table.
25    pub fn new() -> Self {
26        Self {
27            vars: HashMap::new(),
28        }
29    }
30
31    /// Insert or update a named value.
32    pub fn insert(&mut self, name: impl Into<String>, value: f64) {
33        self.vars.insert(name.into(), value);
34    }
35
36    /// Borrow a single expression by name.
37    ///
38    /// The returned `Expr<'_>` borrows from `self` — it cannot outlive this table.
39    pub fn bind(&self, name: &str) -> Option<Expr<'_>> {
40        self.vars.get_key_value(name).map(|(k, &v)| Expr {
41            name: k.as_str(),
42            value: v,
43            unit: infer_unit(k),
44        })
45    }
46
47    /// Iterate over all bound expressions.
48    pub fn bind_all(&self) -> impl Iterator<Item = Expr<'_>> {
49        self.vars.iter().map(|(k, &v)| Expr {
50            name: k.as_str(),
51            value: v,
52            unit: infer_unit(k),
53        })
54    }
55}
56
57impl Default for SymbolTable {
58    fn default() -> Self {
59        Self::new()
60    }
61}
62
63/// Infer a SI unit string from a common variable name.
64fn infer_unit(name: &str) -> &'static str {
65    match name {
66        "velocity" | "speed" | "vx" | "vy" | "v0" => "m/s",
67        "mass" | "m" => "kg",
68        "length" | "l" | "radius" | "r" => "m",
69        "time" | "t" => "s",
70        "angle" | "theta" => "rad",
71        "energy" | "e" => "J",
72        "force" | "f" => "N",
73        "acceleration" | "a" | "g" => "m/s²",
74        _ => "",
75    }
76}

78#[cfg(test)]
79mod tests {
80    use super::*;
81
82    #[test]
83    fn expr_borrows_name_not_cloned() {
84        let mut table = SymbolTable::new();
85        table.insert("velocity", 9.8);
86        let expr = table.bind("velocity").unwrap();
87        assert_eq!(expr.name, "velocity");
88        assert_eq!(expr.value, 9.8);
89        assert_eq!(expr.unit, "m/s");
90    }
91
92    #[test]
93    fn bind_all_yields_all_entries() {
94        let mut table = SymbolTable::new();
95        table.insert("mass", 1.0);
96        table.insert("velocity", 2.0);
97        let count = table.bind_all().count();
98        assert_eq!(count, 2);
99    }
100
101    #[test]
102    fn bind_missing_returns_none() {
103        let table = SymbolTable::new();
104        assert!(table.bind("missing").is_none());
105    }
106}
