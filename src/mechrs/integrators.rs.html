1//! Generic numerical integrators as lazy iterators.
2//!
3//! Both [`euler_iter`] and [`rk4_iter`] return infinite iterators of `(t, y)` pairs
4//! using `std::iter::successors` — no Vec, no heap allocation in the hot path.
5
6/// Euler method iterator for `dy/dt = f(t, y)`.
7///
8/// Simple first-order method. Fast but accumulates error quickly.
9///
10/// # Example
11/// ```
12/// use mechrs::integrators::euler_iter;
13/// let (t, y) = euler_iter(|_, y| y, 0.0, 1.0, 0.001).nth(1000).unwrap();
14/// // y ≈ e^1.0 ≈ 2.718 (with Euler error)
15/// ```
16pub fn euler_iter(
17    f: impl Fn(f64, f64) -> f64 + 'static,
18    t0: f64,
19    y0: f64,
20    dt: f64,
21) -> impl Iterator<Item = (f64, f64)> {
22    std::iter::successors(Some((t0, y0)), move |&(t, y)| {
23        let y_next = y + dt * f(t, y);
24        Some((t + dt, y_next))
25    })
26}
27
28/// Fourth-order Runge-Kutta iterator for `dy/dt = f(t, y)`.
29///
30/// Much more accurate than Euler at the same step size.
31///
32/// # Example
33/// ```
34/// use mechrs::integrators::rk4_iter;
35/// use approx::assert_abs_diff_eq;
36/// let (_, y) = rk4_iter(|_, y| y, 0.0, 1.0, 0.001).nth(1000).unwrap();
37/// assert_abs_diff_eq!(y, 1.0_f64.exp(), epsilon = 1e-6);
38/// ```
39pub fn rk4_iter(
40    f: impl Fn(f64, f64) -> f64 + 'static,
41    t0: f64,
42    y0: f64,
43    dt: f64,
44) -> impl Iterator<Item = (f64, f64)> {
45    std::iter::successors(Some((t0, y0)), move |&(t, y)| {
46        let k1 = f(t, y);
47        let k2 = f(t + dt / 2.0, y + dt / 2.0 * k1);
48        let k3 = f(t + dt / 2.0, y + dt / 2.0 * k2);
49        let k4 = f(t + dt, y + dt * k3);
50        let y_next = y + (dt / 6.0) * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
51        Some((t + dt, y_next))
52    })
53}

55#[cfg(test)]
56mod tests {
57    use super::*;
58
59    #[test]
60    fn rk4_exponential_growth() {
61        // dy/dt = y, y(0) = 1 => y(t) = e^t
62        let (_, y) = rk4_iter(|_, y| y, 0.0, 1.0, 0.001).nth(1000).unwrap();
63        approx::assert_abs_diff_eq!(y, 1.0_f64.exp(), epsilon = 1e-6);
64    }
65
66    #[test]
67    fn rk4_exponential_decay() {
68        // dy/dt = -y, y(0) = 1 => y(1) ≈ 0.3679
69        let (_, y) = rk4_iter(|_, y| -y, 0.0, 1.0, 0.001).nth(1000).unwrap();
70        approx::assert_abs_diff_eq!(y, (-1.0_f64).exp(), epsilon = 1e-6);
71    }
72
73    #[test]
74    fn euler_worse_than_rk4() {
75        let euler_err = {
76            let (_, y) = euler_iter(|_, y| y, 0.0, 1.0, 0.01).nth(100).unwrap();
77            (y - 1.0_f64.exp()).abs()
78        };
79        let rk4_err = {
80            let (_, y) = rk4_iter(|_, y| y, 0.0, 1.0, 0.01).nth(100).unwrap();
81            (y - 1.0_f64.exp()).abs()
82        };
83        assert!(
84            rk4_err < euler_err,
85            "RK4 should be more accurate than Euler"
86        );
87    }
