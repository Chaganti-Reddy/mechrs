1//! Two-body Kepler orbit via Velocity Verlet integration.
2
3use std::ops::{Add, Mul, Sub};
4
5/// 2D vector for orbital mechanics.
6#[derive(Clone, Copy, Debug, PartialEq)]
7pub struct Vec2 {
8    pub x: f64,
9    pub y: f64,
10}
11
12impl Vec2 {
13    pub fn new(x: f64, y: f64) -> Self {
14        Self { x, y }
15    }
16
17    pub fn zero() -> Self {
18        Self { x: 0.0, y: 0.0 }
19    }
20
21    /// Euclidean norm (magnitude).
22    pub fn norm(&self) -> f64 {
23        (self.x * self.x + self.y * self.y).sqrt()
24    }
25
26    /// Unit vector in the same direction.
27    pub fn normalized(&self) -> Self {
28        let n = self.norm();
29        Self {
30            x: self.x / n,
31            y: self.y / n,
32        }
33    }
34
35    /// Dot product.
36    pub fn dot(&self, other: Self) -> f64 {
37        self.x * other.x + self.y * other.y
38    }
39}
40
41impl Add for Vec2 {
42    type Output = Self;
43    fn add(self, rhs: Self) -> Self {
44        Self {
45            x: self.x + rhs.x,
46            y: self.y + rhs.y,
47        }
48    }
49}
50
51impl Sub for Vec2 {
52    type Output = Self;
53    fn sub(self, rhs: Self) -> Self {
54        Self {
55            x: self.x - rhs.x,
56            y: self.y - rhs.y,
57        }
58    }
59}
60
61impl Mul<f64> for Vec2 {
62    type Output = Self;
63    fn mul(self, s: f64) -> Self {
64        Self {
65            x: self.x * s,
66            y: self.y * s,
67        }
68    }
69}

71/// Gravitational acceleration at position `r` for central mass `gm = G*M`.
72fn gravity(gm: f64, r: Vec2) -> Vec2 {
73    let r3 = r.norm().powi(3);
74    r * (-gm / r3)
75}
76
77/// Velocity Verlet orbit iterator.
78///
79/// Yields `(position, velocity)` at each time step.
80/// Velocity Verlet conserves energy much better than Euler for orbital dynamics.
81///
82/// # Arguments
83/// - `gm`: G × central mass (m³/s²)
84/// - `r0`: initial position (m)
85/// - `v0`: initial velocity (m/s)
86/// - `dt`: time step (s)
87pub fn verlet_orbit(gm: f64, r0: Vec2, v0: Vec2, dt: f64) -> impl Iterator<Item = (Vec2, Vec2)> {
88    // State: (position, velocity, acceleration)
89    let a0 = gravity(gm, r0);
90    std::iter::successors(Some((r0, v0, a0)), move |&(r, v, a)| {
91        let r_next = r + v * dt + a * (0.5 * dt * dt);
92        let a_next = gravity(gm, r_next);
93        let v_next = v + (a + a_next) * (0.5 * dt);
94        Some((r_next, v_next, a_next))
95    })
96    .map(|(r, v, _)| (r, v))
97}
98
99/// Specific orbital energy: `E = v²/2 - GM/r`.
100pub fn specific_energy(gm: f64, r: Vec2, v: Vec2) -> f64 {
101    0.5 * v.dot(v) - gm / r.norm()
102}
103
104/// Circular orbit speed at radius `r`: `v = √(GM/r)`.
105pub fn circular_speed(gm: f64, r: f64) -> f64 {
106    (gm / r).sqrt()
107}

109#[cfg(test)]
110mod tests {
111    use super::*;
112
113    const GM_SUN: f64 = 1.327e20; // m³/s²
114    const AU: f64 = 1.496e11; // m
115
116    #[test]
117    fn circular_orbit_constant_radius() {
118        let r = AU;
119        let v = circular_speed(GM_SUN, r);
120        let r0 = Vec2::new(r, 0.0);
121        let v0 = Vec2::new(0.0, v);
122        let dt = 3600.0 * 24.0; // 1 day in seconds
123
124        let radii: Vec<f64> = verlet_orbit(GM_SUN, r0, v0, dt)
125            .take(365)
126            .map(|(pos, _)| pos.norm())
127            .collect();
128
129        let initial = radii[0];
130        for radius in &radii {
131            // Allow 0.1% relative drift (Verlet is symplectic but not exact)
132            approx::assert_abs_diff_eq!(radius / initial, 1.0, epsilon = 1e-3);
133        }
134    }
135
136    #[test]
137    fn energy_conserved_circular() {
138        let r = AU;
139        let v = circular_speed(GM_SUN, r);
140        let r0 = Vec2::new(r, 0.0);
141        let v0 = Vec2::new(0.0, v);
142        let dt = 3600.0 * 24.0;
143        let e0 = specific_energy(GM_SUN, r0, v0);
144
145        let energies: Vec<f64> = verlet_orbit(GM_SUN, r0, v0, dt)
146            .take(365)
147            .map(|(pos, vel)| specific_energy(GM_SUN, pos, vel))
148            .collect();
149
150        for e in &energies {
151            approx::assert_abs_diff_eq!(e, &e0, epsilon = e0.abs() * 1e-6);
152        }
153    }
154
155    #[test]
156    fn vec2_ops() {
157        let a = Vec2::new(3.0, 4.0);
158        assert_eq!(a.norm(), 5.0);
159        let b = Vec2::new(1.0, 0.0);
160        let c = Vec2::new(0.0, 1.0);
161        assert_eq!((b + c), Vec2::new(1.0, 1.0));
162    }
